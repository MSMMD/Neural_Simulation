    \section{Introdução}
    \paragraph{}Buscamos realizar uma simulação simplificada da interação neuronal modelada por um grafo.
        Os neurônios são representados pelos vértices, e as interações de estímulo e inibição pelas arestas.
        Cada neurônio possui uma probabilidade de disparo determinada pelo seu potencial de membrana.
        A simulação ocorre em tempo discreto: um evento de disparo zera o potencial do neurônio disparador
        e modifica o potencial dos neurônios conectados de acordo com as arestas do grafo.

    \paragraph{}Neste momento, o objetivo é encontrar uma função que relaciona a probabilidade de disparo com
    o potencial de membrana. Por razões biológicas, quanto maior o potencial maior a probabilida, logo queremos uma
    função estritamente crescente. Por se tratar de probabilidade, a imagem desta função deverá ser limitada $[0,1]$.

    \paragraph{}Consequentemente, uma sigmoide se apresenta como um notável candidato.\\
    Definiremos $\varphi: \mathbb{R} \rightarrow (0,1)$ como
    \begin{equation*}
        \varphi(x) = \frac{1}{1 + e^{-f(x)}}
    \end{equation*}

    \paragraph{}Com o intuito de simplificar a função, limitaremos $f$ a ser uma função linear $f(x) = x/a - b$.
    Com $f$ desta maneira, $a$ influenciará na inclinação da sigmoide e $b$, seu deslocamento horizontal.
    \paragraph{}Para termos $\varphi(0) = \epsilon$, $b = ln(\frac{1}{\epsilon} - 1)$. Já a inclinação dependerá de quanto o potencial
    de certo neurônio possa variar na passagem do tempo. Sendo $p_{w^+}$ e $p_{w^-}$ os pesos esperados positivo e negativo das arestas,
    $(p_{w^+} - p_{w^-})(N-1)$ representa a variação de potencial esperada em certo neurônio se todos os demais dispararem.
    Então, escolheremos $a \sim (p_{w^+} - p_{w^-})(N-1)$.

    \section{Métodos}
        \subsection{Modelo da Rede Neural baseada em Grafos Orientados}

        \subsubsection{Arquitetura do Grafo}
        \begin{itemize}
            \item \textbf{Vértices:} Usamos $N$ vértices para representar os
            neurônios de maneira homogênea.
            \item \textbf{Arestas:} Usamos um grafo orientado aleatório $G(N, 0.8)$.
            \item \textbf{Peso das arestas:} Representamos as ações de estimulo e
            inibição entre os neurônios. Sendo $w_{i\rightarrow j}$ o peso da aresta que
            conecta o vértice $i$ até o $j$, neste sentido,
            \begin{equation*}
                w_{i \rightarrow j} =
                \begin{cases}
                    +1 \text{ com chance } \frac{50\%}{0.8}\\
                    -1 \text{ com chance } \frac{30\%}{0.8}.
                \end{cases}
            \end{equation*} 
        \end{itemize}

        \subsubsection{Dinâmica do Potencial de Membrana e Disparo}
        \begin{itemize}
            \item \textbf{Potencial de Membrana ($V_i$):} Representamos por um valor real.
            \item \textbf{Probabilidade de Disparo no tempo $t$:} Calculamos a probabilidade segundo uma
            f.d.p. $\varphi: \mathbb{R} \rightarrow (0, 1)$.
            Sendo $\epsilon = \varphi(0)$, $p_{w^+}$ e $p_{w^-}$ os pesos esperados positivo e
            negativo das arestas, respectivamente e o $p_\mathbf{e}$ o valor que desejamos do $\varphi(\bar V)$.
            Definimos $\varphi$ como
                \begin{gather*}
                    \varphi(x) = \frac{1}{1+ e^{-(x/a - b)}}\\
                    b = \ln(1/\epsilon - 1)\\
                    a = \frac{(p_{w^+} - p_{w^-})(N-1)}{p_\mathbf{e}/\epsilon}.
                \end{gather*}
        \end{itemize}

        \subsubsection{Regras de propagação}
            \begin{enumerate}
                \item \textbf{Vazamento do potencial:} Sendo $\rho \in [0,1]$, na passagem do tempo $t$ para $t+1$,
                $V_i \leftarrow \rho \cdot V_i$.
                
                \item \textbf{Modulação do Potencial:} Para cada neurônio vizinho $j$ de um neurônio disparado $i$,
                $V_j \leftarrow V_j + w_{i \rightarrow j}$.

                \item \textbf{Restauração do Potencial:} Caso um neurônio $i$ tenha disparado, $V_i \leftarrow 0$.
            \end{enumerate}

        \subsection{Implementação Computacional}
            \subsubsection{Ambiente de desenvolvimento e linguagem}
            \begin{itemize}
                \item \textbf{Linguagem de programação:} C++17.
                \item \textbf{Compiladores:} MinGw, GCC.
                \item \textbf{Sistema Operacioanl:} Windows 11, Linux(Ubuntu).
            \end{itemize}

            \subsubsection{Estrutura de Dados para o Grafo}
            \begin{itemize}
                \item \textbf{Representação do Grafo:} Armazenamos os pesos por meio de listas de adjacência $viz$, com um
                \emph{array} da estrutura \emph{vector}. Cada elemento na lista do vértice $j$ guarda o \emph{pair}
                $(i, w_{i\rightarrow j})$.

                \item \textbf{Geração de N.A.:} Usamos a biblioteca padrão \lstinline{<random>} e o algoritmo
                \lstinline{mt19937} para fazer a \lstinline{uniform_real_distribution<double> dist(0.0, 1.0);}.
            \end{itemize}

            \subsubsection{Parâmetros da Simulação e Condições Iniciais}
            \begin{itemize}
                \item \textbf{Parâmetros:}
                    \begin{itemize}
                        \item \lstinline{N}: Número de neurônios. $N$ testados: 70, 100, 300 e 800.
                        \item \lstinline{n}: Duração da simulação. $n$ testados: 10, 100, 1000 e 5000.
                        \item \lstinline{rho}: $\rho = 0.95$.
                        \item \lstinline{pmax}: $p_{w^+} = 0.5$.
                        \item \lstinline{pmin}: $p_{w^-} = 0.3$.
                        \item \lstinline{p0}: $\epsilon = 0.05$.
                        \item \lstinline{pi}: $p_{\mathrm{e}} = 0.3$.
                    \end{itemize}

                \item \textbf{Condições Iniciais:}
                    \begin{itemize}
                        \item \lstinline{V[N]}: Potenciais iniciais ($V_i$). Escolhemos $V_i = \varphi^{-1}(p_\mathrm{e})$.
                    \end{itemize}
            \end{itemize}

        \subsection{Análise de Dados e Métricas}
            \begin{itemize}
                \item \textbf{Métricas de Atividade Neural:} Quantidade de disparos no tempo $t$.
                \item \textbf{Métrica do Potencial de Membrana:}
                    \begin{itemize}
                        \item $\bar V$ no tempo $t$.
                        \item $(V_i)_{i \leq N}$ no tempo $n+1$.
                        \item $V_i$ máximo e mínimo em toda simulação
                    \end{itemize}
            \end{itemize}
